# 1. Generate your API key (one-time setup) curl -X POST https://api.webagentbridge.com/v1/keys/generate \ -H "Content-Type: application/json" \ -d '{"plan":"FREE","owner":"Your Name","email":"you@example.com"}' # 2. Use your key to call any WAB module curl -X POST https://api.webagentbridge.com/v1/dark-pattern/analyze \ -H "Authorization: Bearer wab_live_fre_YOUR_KEY" \ -H "Content-Type: application/json" \ -d '{"url":"https://amazon.com/dp/B0EXAMPLE"}'
// Install: npm install @wab/sdk import { WAB } from '@wab/sdk'; const wab = new WAB({ apiKey: 'wab_live_fre_YOUR_KEY' }); // Check a URL for dark patterns const result = await wab.darkPattern.analyze({ url: 'https://example.com/checkout' }); console.log(result.patterns_found); // ['urgency_timer', 'hidden_fees']
# pip install wab-sdk from wab import WABClient client = WABClient(api_key="wab_live_fre_YOUR_KEY") # Scan a URL for scams result = client.scam_shield.scan("https://suspicious-site.com") print(result["threat_level"]) # "HIGH" print(result["confidence"]) # 0.97

Authentication

        
WAB supports three ways to pass your API key:

        

# Option 1: Authorization header (recommended)
Authorization: Bearer wab_live_fre_YOUR_KEY

# Option 2: Custom header
X-WAB-Key: wab_live_fre_YOUR_KEY

# Option 3: Query parameter
GET /v1/dark-pattern/analyze?api_key=wab_live_fre_YOUR_KEY
